788  if (gpio_get_dir(clockPin) != GPIO_OUT) { return MY_INVALID_DIR; }
789  if (gpio_get_dir(dataPin) != GPIO_OUT) { return MY_INVALID_DIR; }
790  for (size_t i=0; i<len; i++) {
791  for (uint8_t j=0; j<8; j++) {
792  uint8_t val = buffer[i];
793  if (bitOrder == MSB_FIRST) {
794  gpio_put(dataPin, (bool)(val & 0x80));
795  val <<= 1;
796  } else {
797  gpio_put(dataPin, (bool)(val & 0x01));
798  val >>= 1;
799  }
800  gpio_put(clockPin, true);
801  gpio_put(clockPin, false);
802  }
803  }
