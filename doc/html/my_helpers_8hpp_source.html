98  int32_t waitForLow32(const uint8_t pin, const uint32_t timeout=0) {
99  uint32_t startTime; // Time started.
100  uint32_t timeoutTime; // absolute timeout time.
101  uint32_t stopTime; // time stopped.
102  // Check pin status:
103  if (isPin(pin) == false) { return MY_INVALID_PIN; }
104  if (gpio_get_dir(pin) == GPIO_OUT) { return MY_INVALID_DIR; }
105  // Check if pin is already low:
106  bool state = gpio_get(pin);
107  if (state == false) { return 0; }
108  // If a timeout is requested wait until timeout.
109  if (timeout > 0) {
110  startTime = time_us_32();
111  timeoutTime = startTime + timeout;
112  while (state == true) {

114  state = gpio_get(pin);
115  }
116  stopTime = time_us_32();
117  // Otherwise block until low.

119  startTime = time_us_32();
120  while (state == true) {
121  state = gpio_get(pin);
122  }
123  stopTime = time_us_32();
124  }
125  // Return the time delta:
