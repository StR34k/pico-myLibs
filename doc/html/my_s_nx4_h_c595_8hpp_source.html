304 void my595::writeByte(const uint8_t value, const bool bitOrder) {
305  if (_useHardware == false) {
306  uint8_t val = value; // copy the value so we can shift it
307  for (uint8_t i=0; i<8; i++) {
308  // Write the bit:
309  if (bitOrder == MSB_FIRST) {
310  gpio_put(_dataTXPin, (bool)(val&0x80));
311  val <<= 1;
312  } else {
313  gpio_put(_dataTXPin, (bool)(val&0x01));
314  val >>= 1;
315  }
316  // Toggle the clock:
317  gpio_put(_clkPin, true);
318  gpio_put(_clkPin, false);
319  }
320  } else {
321  uint8_t val = value;
322  if (bitOrder == LSB_FIRST) {
323  val = myHelpers::reverse8(value);
324  }
325  spi_write_blocking(_spiPort, &val, 1);
326  }
