119 void my595::writeByte(const uint8_t value, const bool bitOrder) {
120  uint8_t val = value; // copy the value so we can shift it
121  for (uint8_t i=0; i<8; i++) {
122  // Write the bit:
123  if (bitOrder == MSB_FIRST) {

125  val <<= 1;
126  } else {

128  val >>= 1;
129  }
