392 int16_t myMCP49x2::initialize(const models model, const bool initSPI) {
393  int16_t returnValue;
394  uint32_t pinMask;
395  _model = model; // Store model #
396 // Init SPI:
397  if (initSPI == true) {
398  if (_useHWSPI == true) {
399  returnValue = mySPI::initializeMaster(_spiPort, _sckPin, _misoPin, _mosiPin, 20000*1000);
400  if (returnValue < 0) { return returnValue; }
401  } else {
402  // Check pins:
403  if (myHelpers::isPin(_sckPin) == false) { return ERROR_INVALID_PIN; }
404  if (myHelpers::isPin(_mosiPin) == false) { return ERROR_INVALID_PIN; }
405  // Add pins to bitMask:
406  pinMask |= (1 << _sckPin) | ( 1 << _mosiPin);
407  }
408  }
409 // Check pins:
410  if (myHelpers::isPin(_csPin) == false) { return ERROR_INVALID_PIN; }
411  pinMask |= (1 << _csPin);
412  if (myHelpers::isPin(_loadPin) == true) { pinMask |= (1 << _loadPin); }
413  if (myHelpers::isPin(_shdnPin) == true) { pinMask |= (1 << _shdnPin); }
414 // Init GPIO:
415  gpio_init_mask(pinMask);
416  gpio_set_dir_out_masked(pinMask);
417  if (_useHWSPI == false) {
418  pinMask &= ~(1 << _sckPin); // Leave pin low.
419  }
420  gpio_put_masked(pinMask, pinMask); // Set pins high.
421  return NO_ERROR;
422 }
