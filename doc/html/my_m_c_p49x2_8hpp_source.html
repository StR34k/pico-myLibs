397 int16_t myMCP49x2::initialize(const models model, const bool initSPI) {
398  int16_t returnValue;
399  uint32_t pinMask;
400  _model = model; // Store model #
401 // Init SPI:
402  if (initSPI == true) {
403  if (_useHWSPI == true) {
404  returnValue = mySPI::initializeMaster(_spiPort, _sckPin, _misoPin, _mosiPin, 20000*1000);
405  if (returnValue < 0) { return returnValue; }
406  } else {
407  // Check pins:
408  if (myHelpers::isPin(_sckPin) == false) { return ERROR_INVALID_PIN; }
409  if (myHelpers::isPin(_mosiPin) == false) { return ERROR_INVALID_PIN; }
410  // Add pins to bitMask:
411  pinMask |= (1 << _sckPin) | ( 1 << _mosiPin);
412  }
413  }
414 // Check pins:
415  if (myHelpers::isPin(_csPin) == false) { return ERROR_INVALID_PIN; }
416  pinMask |= (1 << _csPin);
417  if (myHelpers::isPin(_loadPin) == true) { pinMask |= (1 << _loadPin); }
418  if (myHelpers::isPin(_shdnPin) == true) { pinMask |= (1 << _shdnPin); }
419 // Init GPIO:
420  gpio_init_mask(pinMask);
421  gpio_set_dir_out_masked(pinMask);
422  if (_useHWSPI == false) {
423  pinMask &= ~(1 << _sckPin); // Leave pin low.
424  }
425  gpio_put_masked(pinMask, pinMask); // Set pins high.
426  return NO_ERROR;
427 }
