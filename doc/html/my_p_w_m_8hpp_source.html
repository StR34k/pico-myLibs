553  int32_t setFrequency(const uint8_t slice, const uint32_t freq) {
554  if (isSlice(slice) == false) { return ERROR_INVALID_SLICE; }
555  if (freq == 0 or freq > MY_PWM_DEFAULT_FREQ) { return ERROR_INVALID_FREQUENCY; }
556  uint32_t clock = MY_PWM_DEFAULT_FREQ;
557  if (getPhaseCorrect(slice) == true) { clock = clock / 2; }
558  uint32_t divider16 = clock / freq / 4096 + (clock % (freq * 4096) != 0);
559  if ((divider16 / 16) == 0) { divider16 = 16; }
560  uint32_t wrap = clock * 16 / divider16 / freq - 1;
561  pwm_set_clkdiv_int_frac(slice, divider16/16, (divider16 &0x0F));
562  __setWrap__(slice, wrap);
563  return (int32_t)wrap;
564  }
