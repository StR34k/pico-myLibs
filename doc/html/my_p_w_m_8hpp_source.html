558  int32_t setFrequency(const uint8_t slice, const uint32_t freq) {
559  if (isSlice(slice) == false) { return ERROR_INVALID_SLICE; }
560  if (freq == 0 or freq > MY_PWM_DEFAULT_FREQ) { return ERROR_INVALID_FREQUENCY; }
561  uint32_t clock = MY_PWM_DEFAULT_FREQ;
562  if (getPhaseCorrect(slice) == true) { clock = clock / 2; }
563  uint32_t divider16 = clock / freq / 4096 + (clock % (freq * 4096) != 0);
564  if ((divider16 / 16) == 0) { divider16 = 16; }
565  uint32_t wrap = clock * 16 / divider16 / freq - 1;
566  pwm_set_clkdiv_int_frac(slice, divider16/16, (divider16 &0x0F));
567  __setWrap__(slice, wrap);
568  return (int32_t)wrap;
569  }
