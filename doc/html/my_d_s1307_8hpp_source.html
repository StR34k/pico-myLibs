557 int16_t myDS1307::getDayOfWeek(const uint16_t year, const uint8_t month, const uint8_t day) {
558  int16_t returnValue;
559 // Validate Date for the range this works for.
560  if (year < 1700 or year > 2399) { return ERROR_INVALID_YEAR; }
561  if (month < 1 or month > 12) { return ERROR_INVALID_MONTH; }
562  uint8_t maxDate = DAYS_PER_MONTH[month-1];
563  if (isLeapYear(year) == true) { maxDate += 1; }
564  if (day < 1 or day > maxDate) { return ERROR_INVALID_DATE; }
565 // Calculate century code:
566  uint8_t centuryCode = CENTURY_CODES[((year / 100) - 17)];
567 // Calculate year code:
568  uint8_t yearCode = year - ((year / 100) * 100);
569  yearCode = (yearCode + (yearCode/4)) % 7;
570 // Get month code:
571  uint8_t monthCode = MONTH_CODES[month-1];
572 // Leap year Code:
573  uint8_t leapYearCode = isLeapYear(year);
574 // Calculate day of week.
575  return (int16_t)((yearCode + monthCode + centuryCode + day - leapYearCode) % 7) + 1;
576 }

651  if (returnValue < 0) { return returnValue; }

653 // Switch from 12h mode to 24h mode.
654  if (__is12h__(rawHours) == true and is12h == false) {

661  rawHours = __updateHours__(rawHours, newHours);
662  // Clear the is12h flag.
663  rawHours &= ~MASK_IS_12H;
664 // Switch from 24h mode to 12h mode.
665  } else if (__is12h__(rawHours) == false and is12h == true) {
666  // Convert and set the hours:

671  newHours = __dec_to_bcd__(hours12);
672  if (isPM == true) { newHours |= MASK_IS_PM; }
673  rawHours = __updateHours__(rawHours, newHours);
674  // Set the is12h bit.
