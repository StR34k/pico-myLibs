625  int16_t returnValue;

627  uint8_t newHours;
628  returnValue = __readRegisters__(REG_HOURS, &rawHours, 1);
629  if (returnValue < 0) { return returnValue; }
630  if (__is12h__(rawHours) == is12h) { return NO_ERROR; } // NO change return okay.
631 // Switch from 12h mode to 24h mode.
632  if (__is12h__(rawHours) == true and is12h == false) {
633  // Convert and set the hours:
634  uint8_t hours12 = __bcd_to_dec__(rawHours & MASK_12H_HOURS_BCD);
635  bool isPM = __isPM__(rawHours);
636  uint8_t hours24;
637  myHelpers::convert12hTo24h(hours12, isPM, &hours24);
638  newHours = __dec_to_bcd__(hours24);
639  rawHours = __updateHours__(rawHours, newHours);
640  // Clear the is12h flag.
641  rawHours &= ~MASK_IS_12H;
642 // Switch from 24h mode to 12h mode.
643  } else if (__is12h__(rawHours) == false and is12h == true) {
644  // Convert and set the hours:
645  uint8_t hours24 = __bcd_to_dec__(rawHours & MASK_24H_HOURS_BCD);
646  uint8_t hours12;
647  bool isPM;
648  myHelpers::convert24hTo12h(hours24, &hours12, &isPM);
649  newHours = __dec_to_bcd__(hours12);
650  if (isPM == true) { newHours |= MASK_IS_PM; }
651  rawHours = __updateHours__(rawHours, newHours);
652  // Set the is12h bit.
653  rawHours |= MASK_IS_12H;
654  }
655  returnValue == __writeRegisters__(REG_HOURS, &rawHours, 1);
