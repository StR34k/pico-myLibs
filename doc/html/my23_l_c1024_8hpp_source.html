387 int32_t my23LC1024::initialize(const uint8_t commsMode) {
388  // __breakpoint();
389 // Store comms mode:
390  _commsMode = commsMode;
391 // Do some basic checks:
392  if (commsMode == COMM_MODE_SQI) { // SQI requires hold and sio2 to be defined.
393  if (_holdPin == MY_NOT_A_PIN) { return ERROR_HOLD_NOT_DEFINED; }
394  if (_sio2Pin == MY_NOT_A_PIN) { return ERROR_SIO2_NOT_DEFINED; }
395  }
396 // Initialize pins as GPIO for now.
397  gpio_init(_csPin);
398  gpio_init(_sckPin);
399  gpio_init(_misoPin);
400  gpio_init(_mosiPin);
401 // Set Directions
402  gpio_set_dir(_csPin, GPIO_OUT); // Set cs as output.
403  gpio_set_dir(_sckPin, GPIO_OUT); // Set sck as output.
404  gpio_set_dir(_misoPin, GPIO_IN); // Set miso as input.
405  gpio_set_dir(_mosiPin, GPIO_OUT); // Set mosi as output.
406 // Set initial states:
407  gpio_put(_csPin, true); // Set cs High.
408  gpio_put(_sckPin, false); // set sck Low
409  gpio_put(_mosiPin, false); // set mosi low
410 
411 // Set pin functions, modes, and inital state of hold and sio2:
412  if (_holdPin != MY_NOT_A_PIN) {
413  gpio_init(_holdPin); // Set hold as gpio.
414  gpio_set_dir(_holdPin, GPIO_OUT); // Set hold as output.
415  gpio_put(_holdPin, true); // set hold High.
416  }
417  if (_sio2Pin != MY_NOT_A_PIN) {
418  gpio_init(_sio2Pin); // set sio2 as gpio.
419  gpio_set_dir(_sio2Pin, GPIO_OUT); // set sio2 as output.
420  gpio_put(_sio2Pin, true); // set sio2 High.
421  }
422 // Initialze comms:
423  __resetComms__();
424  // __breakpoint();
425  if (_useHWSPI == true) {
426  spi_init(_spiPort, 1000*20000); // init spi at 20 Mhz.
427  gpio_set_function(_sckPin, GPIO_FUNC_SPI); // set sck as spi.
428  gpio_set_function(_misoPin, GPIO_FUNC_SPI); // set miso as spi.
429  gpio_set_function(_mosiPin, GPIO_FUNC_SPI); // set mosi as spi.
430  } else {
431  uint8_t instruction;
432  switch (_commsMode) {
433  case COMM_MODE_SPI:
434  __setSPIPinModes__();
435  break;
436  case COMM_MODE_SDI:
437  __setSPIPinModes__();
438  __selectChip__();
439  instruction = EDIO_INSTRUCTION;
440  __SPIWrite__(&instruction, 1);
441  __deselectChip__();
442  __setSDIPinModes__(true);
443  break;
444  case COMM_MODE_SQI:
445  __setSPIPinModes__();
446  __selectChip__();
447  instruction = EQIO_INSTRUCTION;
448  __SPIWrite__(&instruction, 1);
449  __deselectChip__();
450  __setSQIPinModes__(true);
451  break;
452  }
453  }
454 // Validate comms by checking mode, setting it something, and checking it again.
455 // Then reset it back to sequential mode.
456  // Read mode:
457  // return true;
458  // __breakpoint();
459  uint8_t mode = __readModeRegister__();
460  uint8_t expectedMode;
461  // Make sure mode is valid:
462  if (mode != SRAM_MODE_BYTE and mode != SRAM_MODE_SEQ and mode != SRAM_MODE_PAGE) {
463  return ERROR_COMM_CHECK_FAILED; // Invalid value returned.
464  }
465  // Write the mode register to a different non zero value. byte mode is 0x00 which could be an error.
466  if (mode != SRAM_MODE_SEQ) {
467  __writeModeRegister__(SRAM_MODE_SEQ);
468  expectedMode = SRAM_MODE_SEQ;
469  } else {
470  __writeModeRegister__(SRAM_MODE_PAGE); // Page mode is non zero value.
471  expectedMode = SRAM_MODE_PAGE;
472  }
473  // Read the mode register and verify:
474  mode = __readModeRegister__();
475  if (mode != expectedMode) {
476  return ERROR_COMM_CHECK_FAILED;
477  }
478  // Set the mode to sequential mode if not already.
479  if (mode != SRAM_MODE_SEQ) {
480  __writeModeRegister__(SRAM_MODE_SEQ);
481  }
482  return NO_ERROR;
483 }

537 int32_t my23LC1024::startRead(const int32_t address) {
538 // Do checks:
539  if (isHeld() == true) { return ERROR_SRAM_HELD; }
540  if (isBusy() == true) { return ERROR_SRAM_BUSY; }
541  if (validateAddress(address) == false) { return ERROR_INVAID_ADDRESS; }
542 // Convert address to an index:
543  int32_t index = address;
544  if (address < 0) { index = LENGTH - abs(address); }
545 // Setup read instructions:
546  uint8_t buffer[4] = {
547  READ_INSTRUCTION, // Send read instruction
548  (uint8_t)((index & ADDRESS_HIGH_BYTE_MASK) >> 16), // Send high byte of address.
549  (uint8_t)((index & ADDRESS_MIDDLE_BYTE_MASK) >> 8), // Send middle byte of address.
550  (uint8_t)(index & ADDRESS_LOW_BYTE_MASK) // Send low byte of address.
551  };
552 // Start Read:
553  __selectChip__();
554  __writeBuffer__(buffer, 4);
555 // Set pin modes and read dummy byte if required:
556  switch (_commsMode) {
557  case COMM_MODE_SDI:
558  // Set pin modes.
559  __setSDIPinModes__(false);
560  // Read dummy byte.
561  __readByte__();
562  break;
563  case COMM_MODE_SQI:
564  // Set pin modes.
565  __setSQIPinModes__(false);
566  // Read dummy byte.
567  __readByte__();
568  break;
569  }
570 // Set the states:
571  __setNextIndex__(index);
572  __setStateRead__();
573  return 0;
574 }

764  sleep_us(1); // wait for a microsecond.

769  for (uint8_t i=0; i<4; i++) {
