167 int32_t my23LC1024::initialize(const uint8_t commsMode) {
168  // __breakpoint();
169 // Store comms mode:
170  _commsMode = commsMode;
171 // Do some basic checks:
172  if (commsMode == COMM_MODE_SQI) { // SQI requires hold and sio2 to be defined.
173  if (_holdPin == MY_NOT_A_PIN) { return ERROR_HOLD_NOT_DEFINED; }
174  if (_sio2Pin == MY_NOT_A_PIN) { return ERROR_SIO2_NOT_DEFINED; }
175  }
176 // Initialize pins as GPIO for now.
177  gpio_init(_csPin);
178  gpio_init(_sckPin);
179  gpio_init(_misoPin);
180  gpio_init(_mosiPin);
181 // Set Directions
182  gpio_set_dir(_csPin, GPIO_OUT); // Set cs as output.
183  gpio_set_dir(_sckPin, GPIO_OUT); // Set sck as output.
184  gpio_set_dir(_misoPin, GPIO_IN); // Set miso as input.
185  gpio_set_dir(_mosiPin, GPIO_OUT); // Set mosi as output.
186 // Set initial states:
187  gpio_put(_csPin, true); // Set cs High.
188  gpio_put(_sckPin, false); // set sck Low
189  gpio_put(_mosiPin, false); // set mosi low
190 
191 // Set pin functions, modes, and inital state of hold and sio2:
192  if (_holdPin != MY_NOT_A_PIN) {
193  gpio_init(_holdPin); // Set hold as gpio.
194  gpio_set_dir(_holdPin, GPIO_OUT); // Set hold as output.
195  gpio_put(_holdPin, true); // set hold High.
196  }
197  if (_sio2Pin != MY_NOT_A_PIN) {
198  gpio_init(_sio2Pin); // set sio2 as gpio.
199  gpio_set_dir(_sio2Pin, GPIO_OUT); // set sio2 as output.
200  gpio_put(_sio2Pin, true); // set sio2 High.
201  }
202 // Initialze comms:
203  __resetComms__();
204  // __breakpoint();
205  if (_useHWSPI == true) {
206  spi_init(_spiPort, 1000*20000); // init spi at 20 Mhz.
207  gpio_set_function(_sckPin, GPIO_FUNC_SPI); // set sck as spi.
208  gpio_set_function(_misoPin, GPIO_FUNC_SPI); // set miso as spi.
209  gpio_set_function(_mosiPin, GPIO_FUNC_SPI); // set mosi as spi.
210  } else {
211  uint8_t instruction;
212  switch (_commsMode) {
213  case COMM_MODE_SPI:
214  __setSPIPinModes__();
215  break;
216  case COMM_MODE_SDI:
217  __setSPIPinModes__();
218  __selectChip__();
219  instruction = EDIO_INSTRUCTION;
220  __SPIWrite__(&instruction, 1);
221  __deselectChip__();
222  __setSDIPinModes__(true);
223  break;
224  case COMM_MODE_SQI:
225  __setSPIPinModes__();
226  __selectChip__();
227  instruction = EQIO_INSTRUCTION;
228  __SPIWrite__(&instruction, 1);
229  __deselectChip__();
230  __setSQIPinModes__(true);
231  break;
232  }
233  }
234 // Validate comms by checking mode, setting it something, and checking it again.
235 // Then reset it back to sequential mode.
236  // Read mode:
237  // return true;
238  // __breakpoint();
239  uint8_t mode = __readModeRegister__();
240  uint8_t expectedMode;
241  // Make sure mode is valid:
242  if (mode != SRAM_MODE_BYTE and mode != SRAM_MODE_SEQ and mode != SRAM_MODE_PAGE) {
243  return ERROR_COMM_CHECK_FAILED; // Invalid value returned.
244  }
245  // Write the mode register to a different non zero value. byte mode is 0x00 which could be an error.
246  if (mode != SRAM_MODE_SEQ) {
247  __writeModeRegister__(SRAM_MODE_SEQ);
248  expectedMode = SRAM_MODE_SEQ;
249  } else {
250  __writeModeRegister__(SRAM_MODE_PAGE); // Page mode is non zero value.
251  expectedMode = SRAM_MODE_PAGE;
252  }
253  // Read the mode register and verify:
254  mode = __readModeRegister__();
255  if (mode != expectedMode) {
256  return ERROR_COMM_CHECK_FAILED;
257  }
258  // Set the mode to sequential mode if not already.
259  if (mode != SRAM_MODE_SEQ) {
260  __writeModeRegister__(SRAM_MODE_SEQ);
261  }
262  return NO_ERROR;
263 }

317 int32_t my23LC1024::startRead(const int32_t address) {
318 // Do checks:
319  if (isHeld() == true) { return ERROR_SRAM_HELD; }
320  if (isBusy() == true) { return ERROR_SRAM_BUSY; }
321  if (validateAddress(address) == false) { return ERROR_INVAID_ADDRESS; }
322 // Convert address to an index:
323  int32_t index = address;
324  if (address < 0) { index = LENGTH - abs(address); }
325 // Setup read instructions:
326  uint8_t buffer[4] = {
327  READ_INSTRUCTION, // Send read instruction
328  (uint8_t)((index & ADDRESS_HIGH_BYTE_MASK) >> 16), // Send high byte of address.
329  (uint8_t)((index & ADDRESS_MIDDLE_BYTE_MASK) >> 8), // Send middle byte of address.
330  (uint8_t)(index & ADDRESS_LOW_BYTE_MASK) // Send low byte of address.
331  };
332 // Start Read:
333  __selectChip__();
334  __writeBuffer__(buffer, 4);
335 // Set pin modes and read dummy byte if required:
336  switch (_commsMode) {
337  case COMM_MODE_SDI:
338  // Set pin modes.
339  __setSDIPinModes__(false);
340  // Read dummy byte.
341  __readByte__();
342  break;
343  case COMM_MODE_SQI:
344  // Set pin modes.
345  __setSQIPinModes__(false);
346  // Read dummy byte.
347  __readByte__();
348  break;
349  }
350 // Set the states:
351  __setNextIndex__(index);
352  __setStateRead__();
353  return 0;
354 }
